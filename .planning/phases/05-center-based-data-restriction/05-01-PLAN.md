3. `filterBundlesByCenters(bundles, userCenters: string[]): FhirBundle[]` — For each bundle:

**Router endpoint:**
```typescript
export const fhirApiRouter = Router();

fhirApiRouter.get('/bundles', async (req: Request, res: Response): Promise<void> => {
  const { role, centers } = req.auth!;
  try {
    const allBundles = await getCachedBundles();
    if (isBypass(role, centers)) {
      res.json({ bundles: allBundles });
      return;
    }
    const filtered = filterBundlesByCenters(allBundles, centers);
    res.json({ bundles: filtered });
  } catch (err) {
    console.error('[fhir-api] Error loading bundles:', (err as Error).message);
    res.status(502).json({ error: 'Failed to load FHIR bundles' });
  }
});
```

const SHORTHAND_TO_ORG: Record<string, string> = {
  'UKA': 'org-uka', 'UKB': 'org-ukb', 'LMU': 'org-lmu',
  'UKT': 'org-ukt', 'UKM': 'org-ukm',
};

function _migrateCenterIds(users: UserRecord[]): { users: UserRecord[]; changed: boolean } {
  let changed = false;
  const migrated = users.map(u => {
    const newCenters = u.centers.map(c => SHORTHAND_TO_ORG[c] ?? c);
    if (newCenters.some((nc, i) => nc !== u.centers[i])) changed = true;
    return { ...u, centers: newCenters };
  });
  return { users: migrated, changed };
}

function validateCaseCenters(caseIds: string[], userCenters: string[], role: string): string | null {
  if (role === 'admin' || userCenters.length >= 5) return null; // bypass
  const index = getCaseToCenter();
  for (const caseId of caseIds) {
    const caseCenterId = index.get(caseId);
    // If caseId is not in the index, allow it (case may not be loaded yet)
    if (caseCenterId && !userCenters.includes(caseCenterId)) {
      return `Case ${caseId} not in user's permitted centers`;
    }
  }
  return null;
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client -> /api/fhir/bundles | Untrusted client requests FHIR data; server must filter by JWT centers |
| client -> /api/data/* writes | Client submits caseIds in quality flags / saved searches; server must validate center ownership |
| JWT payload -> req.auth | JWT is signed server-side; centers array is trusted after authMiddleware verification |
| settings change -> FHIR cache | Admin changes data source; stale cache must be invalidated |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Information Disclosure | /api/fhir/bundles | mitigate | filterBundlesByCenters() removes all bundles/resources not in req.auth.centers before response; bypass only for admin or all-centers users |
| T-05-02 | Tampering | PUT /quality-flags | mitigate | validateCaseCenters() checks caseId -> centerId mapping against req.auth.centers; 403 on mismatch |
| T-05-03 | Spoofing | JWT centers array | accept | authMiddleware validates JWT signature with server secret; centers come from signed token, not user input |
| T-05-04 | Information Disclosure | FHIR cache (stale after settings change) | mitigate | invalidateFhirCache() imported and called in server/settingsApi.ts after successful PUT /api/settings write — both Vite plugin and production handler paths |
| T-05-05 | Elevation of Privilege | Bypass via large centers array | accept | centers.length >= VALID_CENTERS.size is safe because centers in JWT are set by server during login from users.json; user cannot inject extra centers |
| T-05-06 | Information Disclosure | Blaze proxy error messages | mitigate | Return generic "Blaze server unreachable" (502), not raw Blaze error details |
</threat_model>
