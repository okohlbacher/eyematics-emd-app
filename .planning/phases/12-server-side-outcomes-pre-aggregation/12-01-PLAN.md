<objective>
Extract pure math + pure FHIR constants + FHIR domain types from `src/` into a top-level `shared/` directory, update both tsconfigs to include it, and install backward-compatibility re-export shims so every existing client-side import keeps working. This is the single precondition that unlocks Plan 12-02's server handler — without the split, a server import of `computeCohortTrajectory` transitively loads `authHeaders.ts` and crashes on `sessionStorage is not defined` (RESEARCH.md §Common Pitfalls #5, §Pattern 1).

Purpose: AGG-02 demands byte-identical client/server aggregation. A shared TS module is the only construction that enforces parity by definition (D-01). Duplicate code + round-trip test was explicitly rejected as drift-prone.

Output: A `shared/` tree Node can import safely, both tsconfigs covering it, zero production-code behavior change for the client, and one Wave-0 parity test proving the move is lossless.
</objective>

      Eye,
      GridPoint,
      PatientSeries,
      TrajectoryResult,
    } from './cohortTrajectory';

      patientCount: number;
      excludedCount: number;
      measurementCount: number;
      cacheHit: boolean;
    }

    export interface AggregateResponse {
      median: GridPoint[];
      iqrLow: number[];
      iqrHigh: number[];
      perPatient?: PatientSeries[];
      scatter?: Array<{ x: number; y: number; patientId: string }>;
      meta: AggregateMeta;
    }

    /**
     * Project TrajectoryResult → AggregateResponse with literal key order.
     * - median first, then iqrLow, then iqrHigh
     * - perPatient and scatter ONLY emitted when the corresponding flag is true
     *   (absent keys are preferable to "undefined" values for JSON.stringify parity)
     * - meta LAST with patientCount → excludedCount → measurementCount → cacheHit
     */
    export function shapeOutcomesResponse(
      trajectory: TrajectoryResult,
      eye: Eye,
      includePerPatient: boolean,
      includeScatter: boolean,
      cacheHit: boolean,
    ): AggregateResponse {
      const panel =
        eye === 'od' ? trajectory.od : eye === 'os' ? trajectory.os : trajectory.combined;

      const result: AggregateResponse = {
        median: panel.medianGrid,
        iqrLow: panel.medianGrid.map((g) => g.p25),
        iqrHigh: panel.medianGrid.map((g) => g.p75),
        meta: {
          patientCount: panel.summary.patientCount,
          excludedCount: panel.summary.excludedCount,
          measurementCount: panel.summary.measurementCount,
          cacheHit,
        },
      };
      if (includePerPatient) result.perPatient = panel.patients;
      if (includeScatter) result.scatter = panel.scatterPoints;
      return result;
    }
