    ```typescript
    // ---------------------------------------------------------------------------
    // computeCrtTrajectory — METRIC-01 / Phase 13 D-03
    // Mirrors computeCohortTrajectory but reads LOINC_CRT observations whose
    // valueQuantity.value is already in µm (no logMAR conversion).
    // ---------------------------------------------------------------------------

    export function computeCrtTrajectory(input: {
      cases: PatientCase[];
      axisMode: AxisMode;
      yMetric: YMetric;
      gridPoints: number;
      spreadMode?: SpreadMode;
    }): TrajectoryResult {
      const { cases, axisMode, yMetric, gridPoints, spreadMode = 'iqr' } = input;

      const odSeries: PatientSeries[] = [];
      const osSeries: PatientSeries[] = [];

      for (const c of cases) {
        const allCrt = getObservationsByCode(c.observations, LOINC_CRT);
        const odObs = allCrt.filter((o) => eyeOf(o.bodySite) === 'od');
        const osObs = allCrt.filter((o) => eyeOf(o.bodySite) === 'os');

        odSeries.push(buildCrtPatientSeries(c, odObs, 'od', axisMode, yMetric, gridPoints));
        osSeries.push(buildCrtPatientSeries(c, osObs, 'os', axisMode, yMetric, gridPoints));
      }

      const combinedSeries: PatientSeries[] = cases.map((c, i) => {
        const od = odSeries[i];
        const os = osSeries[i];
        const allMeas = [...od.measurements, ...os.measurements].sort((a, b) => a.x - b.x);
        const included = !od.excluded || !os.excluded;
        return {
          id: c.pseudonym,
          pseudonym: c.pseudonym,
          measurements: allMeas,
          sparse: allMeas.length >= 2 && allMeas.length < Math.ceil(gridPoints / 10),
          excluded: !included,
          baseline: od.baseline ?? os.baseline,
        };
      });

      return {
        od: buildPanel(odSeries, gridPoints, spreadMode),
        os: buildPanel(osSeries, gridPoints, spreadMode),
        combined: buildPanel(combinedSeries, gridPoints, spreadMode),
      };
    }

    function buildCrtPatientSeries(
      c: PatientCase,
      crtObs: Observation[],
      eye: 'od' | 'os',
      axisMode: AxisMode,
      yMetric: YMetric,
      gridPoints: number,
    ): PatientSeries {
      if (crtObs.length === 0) {
        return { id: c.pseudonym, pseudonym: c.pseudonym, measurements: [], sparse: false, excluded: true, baseline: undefined };
      }

      // Sort ascending by date; baseline = earliest µm value
      const sorted = [...crtObs]
        .filter((o) => typeof o.effectiveDateTime === 'string' && typeof o.valueQuantity?.value === 'number' && Number.isFinite(o.valueQuantity.value))
        .sort((a, b) => (a.effectiveDateTime as string).localeCompare(b.effectiveDateTime as string));

      if (sorted.length === 0) {
        return { id: c.pseudonym, pseudonym: c.pseudonym, measurements: [], sparse: false, excluded: true, baseline: undefined };
      }

      const baselineDate = new Date(sorted[0].effectiveDateTime as string);
      const baselineUm = sorted[0].valueQuantity!.value as number;

      // Pre-collect IVI procedure dates for treatmentIndex (same as visus)
      const iviDatesForEye: string[] = (c.procedures ?? [])
        .filter((p) => (p.code?.coding ?? []).some((cd) => cd.code === SNOMED_IVI))
        .filter((p) => eyeOf(p.bodySite) === eye)
        .map((p) => p.performedDateTime)
        .filter((d): d is string => typeof d === 'string')
        .sort();

      const measurements = sorted.map((o) => {
        const obsDate = new Date(o.effectiveDateTime as string);
        const days = Math.round((obsDate.getTime() - baselineDate.getTime()) / 86400000);
        const x = axisMode === 'days' ? days : iviDatesForEye.filter((d) => d <= (o.effectiveDateTime as string)).length;
        const raw = o.valueQuantity!.value as number;
        let y: number;
        if (yMetric === 'absolute') {
          y = raw;
        } else if (yMetric === 'delta') {
          y = raw - baselineUm;
        } else {
          // delta_percent — clamp to ±200 (matches visus)
          const pct = baselineUm === 0 ? 0 : ((raw - baselineUm) / baselineUm) * 100;
          y = Math.max(-200, Math.min(200, pct));
        }
        return { x, y, rawDecimal: raw, obsDate: o.effectiveDateTime as string };
      });

      return {
        id: c.pseudonym,
        pseudonym: c.pseudonym,
        measurements,
        sparse: measurements.length >= 2 && measurements.length < Math.ceil(gridPoints / 10),
        excluded: false,
        baseline: { date: baselineDate.toISOString().slice(0, 10), value: baselineUm },
      };
    }
    ```
