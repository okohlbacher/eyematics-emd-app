    interface CrtRow {
      patient_pseudonym: string;
      eye: 'od' | 'os';
      observation_date: string;
      days_since_baseline: number;
      crt_um: number;
      crt_delta_um: number;
    }

    function flattenCrtRows(cases: PatientCase[]): CrtRow[] {
      const rows: CrtRow[] = [];
      for (const pc of cases) {
        const crtByEye: Record<'od' | 'os', Array<{ date: string; um: number }>> = { od: [], os: [] };
        for (const obs of pc.observations ?? []) {
          const isCrt = (obs.code?.coding ?? []).some((c) => c.code === LOINC_CRT);
          if (!isCrt) continue;
          const e = eyeOf(obs.bodySite);
          if (e !== 'od' && e !== 'os') continue;
          const um = typeof obs.valueQuantity?.value === 'number' ? obs.valueQuantity.value : NaN;
          if (!Number.isFinite(um)) continue;
          const date = typeof obs.effectiveDateTime === 'string' ? obs.effectiveDateTime.slice(0, 10) : '';
          if (!date) continue;
          crtByEye[e].push({ date, um });
        }
        (['od', 'os'] as const).forEach((eye) => crtByEye[eye].sort((a, b) => a.date.localeCompare(b.date)));
        (['od', 'os'] as const).forEach((eye) => {
          const obs = crtByEye[eye];
          if (obs.length === 0) return;
          const baseline = obs[0];
          for (const m of obs) {
            const days = Math.round((new Date(m.date).getTime() - new Date(baseline.date).getTime()) / 86400000);
            rows.push({
              patient_pseudonym: pc.pseudonym,
              eye,
              observation_date: m.date,
              days_since_baseline: days,
              crt_um: m.um,
              crt_delta_um: m.um - baseline.um,
            });
          }
        });
      }
      return rows;
    }

    interface IntervalRow {
      patient_pseudonym: string;
      eye: 'od' | 'os';
      gap_index: number;
      gap_days: number;
      procedure_date: string;
    }

    function flattenIntervalRows(cases: PatientCase[]): IntervalRow[] {
      const rows: IntervalRow[] = [];
      for (const pc of cases) {
        (['od', 'os'] as const).forEach((eye) => {
          const dates = (pc.procedures ?? [])
            .filter((p) => (p.code?.coding ?? []).some((c) => c.code === SNOMED_IVI))
            .filter((p) => eyeOf(p.bodySite) === eye)
            .map((p) => p.performedDateTime)
            .filter((d): d is string => typeof d === 'string')
            .sort();
          for (let i = 1; i < dates.length; i++) {
            const gap = Math.floor((new Date(dates[i]).getTime() - new Date(dates[i - 1]).getTime()) / 86400000);
            if (gap < 0) continue;
            rows.push({
              patient_pseudonym: pc.pseudonym,
              eye,
              gap_index: i,
              gap_days: gap,
              procedure_date: dates[i].slice(0, 10),
            });
          }
        });
      }
      return rows;
    }

    interface ResponderRow {
      patient_pseudonym: string;
      eye: 'combined';
      bucket: 'responder' | 'partial' | 'nonResponder';
      delta_visus_letters: number;
      measurement_date: string;
    }

    function flattenResponderRows(cases: PatientCase[], thresholdLetters: number): ResponderRow[] {
      const buckets = classifyResponders(cases, thresholdLetters, 'combined');
      const rows: ResponderRow[] = [];
      const addBucket = (bucket: ResponderRow['bucket'], list: PatientCase[]) => {
        for (const pc of list) {
          // Recompute delta + measurement date for the row. Use the same day-365 search as classifyResponders.
          // Simplification: find the visus observation closest to baseline+365 within ±180d.
          // Since classifyResponders already verified existence, just compute here.
          const visus = (pc.observations ?? [])
            .filter((o) => (o.code?.coding ?? []).some((c) => c.code === LOINC_VISUS))
            .map((o) => ({ date: o.effectiveDateTime as string, decimal: o.valueQuantity?.value as number }))
            .filter((m) => typeof m.date === 'string' && Number.isFinite(m.decimal) && m.decimal > 0)
            .sort((a, b) => a.date.localeCompare(b.date));
          if (visus.length < 2) continue;
          const baseline = visus[0];
          const baselineMs = new Date(baseline.date).getTime();
          let best: { date: string; decimal: number; delta: number } | null = null;
          for (const v of visus.slice(1)) {
            const days = (new Date(v.date).getTime() - baselineMs) / 86400000;
            if (Math.abs(days - 365) > 180) continue;
            if (!best || Math.abs(days - 365) < Math.abs((new Date(best.date).getTime() - baselineMs) / 86400000 - 365)) {
              best = { date: v.date, decimal: v.decimal, delta: 0 };
            }
          }
          if (!best) continue;
          const logmarBase = -Math.log10(baseline.decimal);
          const logmarYr1 = -Math.log10(best.decimal);
          const deltaLetters = Math.round((logmarBase - logmarYr1) * 50); // positive = improvement
          rows.push({
            patient_pseudonym: pc.pseudonym,
            eye: 'combined',
            bucket,
            delta_visus_letters: deltaLetters,
            measurement_date: best.date.slice(0, 10),
          });
        }
      };
      addBucket('responder', buckets.responder);
      addBucket('partial', buckets.partial);
      addBucket('nonResponder', buckets.nonResponder);
      return rows;
    }
