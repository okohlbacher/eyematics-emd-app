  if (!user) return;  // Don't fetch if not authenticated (Pitfall 2)

    const headers = getAuthHeaders();
    const [flagsResp, searchesResp, excludedResp, reviewedResp] = await Promise.all([

      fetch('/api/data/saved-searches', { headers }),

      fetch('/api/data/reviewed-cases', { headers }),

      flagsResp.json() as Promise<{ qualityFlags: QualityFlag[] }>,
      searchesResp.json() as Promise<{ savedSearches: SavedSearch[] }>,

    setReviewedCases(reviewedData.reviewedCases ?? []);
    setDataLoading(false);
