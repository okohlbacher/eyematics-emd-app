  const errors: string[] = [];

    setQualityFlags(results[0].value.qualityFlags ?? []);
  } else {
    errors.push(results[0].reason?.message ?? 'quality-flags failed');
  }

  if (results[1].status === 'fulfilled') {

  } else {

  }
  setDataLoading(false);
}, [user]);
