```typescript
/**
 * Atomically write the full user list to data/users.json.
 * Uses temp-file + rename to prevent corruption on crash.
 */
export function saveUsers(users: UserRecord[]): void {
  if (_usersFile === null) {
    throw new Error('[initAuth] saveUsers() called before initAuth()');
  }
  _atomicWrite(_usersFile, JSON.stringify(users, null, 2));
}
```

```typescript
authApiRouter.post('/users', (req: Request, res: Response): void => {
  if (!req.auth || req.auth.role !== 'admin') {
    res.status(403).json({ error: 'Admin access required' });
    return;
  }

  const { username, role, centers, firstName, lastName } = req.body as Record<string, unknown>;

  // Validate required fields
  if (typeof username !== 'string' || !username.trim()) {
    res.status(400).json({ error: 'username is required' });
    return;
  }
  const validRoles = ['admin', 'researcher', 'epidemiologist', 'clinician', 'data_manager', 'clinic_lead'];
  const userRole = typeof role === 'string' && validRoles.includes(role) ? role : 'researcher';
  const userCenters = Array.isArray(centers) ? centers.filter((c): c is string => typeof c === 'string') : [];

  // Check duplicate
  const users = loadUsers();
  if (users.find((u) => u.username.toLowerCase() === username.trim().toLowerCase())) {
    res.status(409).json({ error: 'Username already exists' });
    return;
  }

  // Generate secure random password (D-01): 16 chars base64url = ~96 bits entropy
  const generatedPassword = crypto.randomBytes(12).toString('base64url').slice(0, 16);
  const passwordHash = bcrypt.hashSync(generatedPassword, 12);

  const newUser = {
    username: username.trim(),
    passwordHash,
    role: userRole,
    centers: userCenters,
    firstName: typeof firstName === 'string' && firstName.trim() ? firstName.trim() : undefined,
    lastName: typeof lastName === 'string' && lastName.trim() ? lastName.trim() : undefined,
    createdAt: new Date().toISOString(),
  };

  users.push(newUser);
  saveUsers(users);

  // Return user WITHOUT passwordHash, plus the one-time generatedPassword
  const { passwordHash: _omit, ...safeUser } = newUser;
  res.status(201).json({ user: safeUser, generatedPassword });
});
```

**4. Add DELETE /users/:username (USER-04, D-03):**

```typescript
authApiRouter.delete('/users/:username', (req: Request, res: Response): void => {
  if (!req.auth || req.auth.role !== 'admin') {
    res.status(403).json({ error: 'Admin access required' });
    return;
  }

  const target = req.params.username;

  // Self-delete guard (Pitfall 6)
  if (req.auth.preferred_username.toLowerCase() === target.toLowerCase()) {
    res.status(409).json({ error: 'Cannot delete your own account' });
    return;
  }

  const users = loadUsers();
  const idx = users.findIndex((u) => u.username.toLowerCase() === target.toLowerCase());
  if (idx === -1) {
    res.status(404).json({ error: 'User not found' });
    return;
  }

  users.splice(idx, 1);
  saveUsers(users);
  res.json({ message: 'User deleted' });
});
```

**5. Add PUT /users/:username/password (USER-11, D-03):**

```typescript
authApiRouter.put('/users/:username/password', (req: Request, res: Response): void => {
  if (!req.auth || req.auth.role !== 'admin') {
    res.status(403).json({ error: 'Admin access required' });
    return;
  }

  const target = req.params.username;
  const { password } = req.body as { password?: string };

  if (typeof password !== 'string' || password.length < 8) {
    res.status(400).json({ error: 'Password must be at least 8 characters' });
    return;
  }

  const users = loadUsers();
  const user = users.find((u) => u.username.toLowerCase() === target.toLowerCase());
  if (!user) {
    res.status(404).json({ error: 'User not found' });
    return;
  }

  user.passwordHash = bcrypt.hashSync(password, 12);
  saveUsers(users);
  res.json({ message: 'Password updated' });
});
```

IMPORTANT: Do NOT log the generated password anywhere. The plaintext is returned in the POST response body only and never persisted or printed to stdout.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client -> API | Untrusted input from browser; all user-supplied data (username, role, centers, password) must be validated server-side |
| admin -> user data | Admin can create/delete users; must not be able to self-delete (loss of admin access) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Elevation of Privilege | POST /users | mitigate | Admin-role check: `req.auth.role !== 'admin'` returns 403; role validated against allowlist |
| T-04-02 | Information Disclosure | POST /users response | mitigate | passwordHash never returned; generatedPassword returned once in 201 response only; never logged |
| T-04-03 | Denial of Service | DELETE /users/:username (self) | mitigate | Self-delete guard: `req.auth.preferred_username === target` returns 409 |
| T-04-04 | Tampering | POST /users username | mitigate | Username validated as non-empty string; duplicate check (case-insensitive) returns 409 |
| T-04-05 | Information Disclosure | GET /users (non-admin) | mitigate | Admin-role guard added to existing GET /users endpoint (was missing per Pitfall 1) |
| T-04-06 | Tampering | SQLite injection via username | mitigate | No SQL in this plan — users.json uses JSON.stringify; data plan (02) uses named parameters |
