Create `server/dataDb.ts` following the exact same pattern as `server/auditDb.ts`. This file manages `data/data.db` (D-07, D-08 — separate from audit.db).

```typescript
/**
 * Data persistence layer — per-user SQLite storage for quality flags,
 * saved searches, excluded cases, and reviewed cases.
 *
 * Mirrors auditDb.ts patterns: synchronous better-sqlite3, WAL mode,
 * named parameters in all queries.
 *
 * Per D-06: all data keyed by username for per-user isolation.
 * Per D-07: separate data.db from audit.db.
 * Per D-08: different access patterns (mutable CRUD vs immutable append-only).
 */

import Database from 'better-sqlite3';
import path from 'node:path';

// ---------------------------------------------------------------------------
// Module-level db instance (populated by initDataDb)
// ---------------------------------------------------------------------------

let db: Database.Database | null = null;

function getDb(): Database.Database {
  if (!db) throw new Error('[dataDb] called before initDataDb()');
  return db;
}

// ---------------------------------------------------------------------------
// 1. initDataDb
// ---------------------------------------------------------------------------

export function initDataDb(dataDir: string): void {
  const dbPath = path.join(dataDir, 'data.db');
  db = new Database(dbPath);
  db.pragma('journal_mode = WAL');

  db.exec(`
    CREATE TABLE IF NOT EXISTS quality_flags (
      username    TEXT NOT NULL,
      case_id     TEXT NOT NULL,
      parameter   TEXT NOT NULL,
      error_type  TEXT NOT NULL,
      flagged_at  TEXT NOT NULL,
      flagged_by  TEXT NOT NULL,
      status      TEXT NOT NULL DEFAULT 'open',
      PRIMARY KEY (username, case_id, parameter)
    );
    CREATE INDEX IF NOT EXISTS idx_qf_username ON quality_flags(username);

    CREATE TABLE IF NOT EXISTS saved_searches (
      id          TEXT PRIMARY KEY,
      username    TEXT NOT NULL,
      name        TEXT NOT NULL,
      created_at  TEXT NOT NULL,
      filters     TEXT NOT NULL
    );
    CREATE INDEX IF NOT EXISTS idx_ss_username ON saved_searches(username);

    CREATE TABLE IF NOT EXISTS excluded_cases (
      username    TEXT NOT NULL,
      case_id     TEXT NOT NULL,
      PRIMARY KEY (username, case_id)
    );

    CREATE TABLE IF NOT EXISTS reviewed_cases (
      username    TEXT NOT NULL,
      case_id     TEXT NOT NULL,
      PRIMARY KEY (username, case_id)
    );
  `);

  console.log('[dataDb] Opened data database:', dbPath);
}

// ---------------------------------------------------------------------------
// 2. Quality Flags
// ---------------------------------------------------------------------------

export interface QualityFlagRow {
  case_id: string;
  parameter: string;
  error_type: string;
  flagged_at: string;
  flagged_by: string;
  status: string;
}

export function getQualityFlags(username: string): QualityFlagRow[] {
  return getDb()
    .prepare('SELECT case_id, parameter, error_type, flagged_at, flagged_by, status FROM quality_flags WHERE username = @username')
    .all({ username }) as QualityFlagRow[];
}

export function setQualityFlags(username: string, flags: QualityFlagRow[]): void {
  const d = getDb();
  const del = d.prepare('DELETE FROM quality_flags WHERE username = @username');
  const ins = d.prepare(
    'INSERT INTO quality_flags (username, case_id, parameter, error_type, flagged_at, flagged_by, status) VALUES (@username, @case_id, @parameter, @error_type, @flagged_at, @flagged_by, @status)'
  );
  const tx = d.transaction(() => {
    del.run({ username });
    for (const f of flags) {
      ins.run({ username, ...f });
    }
  });
  tx();
}

// ---------------------------------------------------------------------------
// 3. Saved Searches
// ---------------------------------------------------------------------------

export interface SavedSearchRow {
  id: string;
  name: string;
  created_at: string;
  filters: string;  // JSON string
}

export function getSavedSearches(username: string): SavedSearchRow[] {
  return getDb()
    .prepare('SELECT id, name, created_at, filters FROM saved_searches WHERE username = @username ORDER BY created_at DESC')
    .all({ username }) as SavedSearchRow[];
}

export function addSavedSearch(username: string, search: SavedSearchRow): void {
  getDb()
    .prepare('INSERT OR REPLACE INTO saved_searches (id, username, name, created_at, filters) VALUES (@id, @username, @name, @created_at, @filters)')
    .run({ username, ...search });
}

export function removeSavedSearch(username: string, id: string): void {
  getDb()
    .prepare('DELETE FROM saved_searches WHERE username = @username AND id = @id')
    .run({ username, id });
}

// ---------------------------------------------------------------------------
// 4. Excluded Cases
// ---------------------------------------------------------------------------

export function getExcludedCases(username: string): string[] {
  const rows = getDb()
    .prepare('SELECT case_id FROM excluded_cases WHERE username = @username')
    .all({ username }) as { case_id: string }[];
  return rows.map((r) => r.case_id);
}

export function setExcludedCases(username: string, caseIds: string[]): void {
  const d = getDb();
  const del = d.prepare('DELETE FROM excluded_cases WHERE username = @username');
  const ins = d.prepare('INSERT INTO excluded_cases (username, case_id) VALUES (@username, @case_id)');
  const tx = d.transaction(() => {
    del.run({ username });
    for (const case_id of caseIds) {
      ins.run({ username, case_id });
    }
  });
  tx();
}

// ---------------------------------------------------------------------------
// 5. Reviewed Cases
// ---------------------------------------------------------------------------

export function getReviewedCases(username: string): string[] {
  const rows = getDb()
    .prepare('SELECT case_id FROM reviewed_cases WHERE username = @username')
    .all({ username }) as { case_id: string }[];
  return rows.map((r) => r.case_id);
}

export function setReviewedCases(username: string, caseIds: string[]): void {
  const d = getDb();
  const del = d.prepare('DELETE FROM reviewed_cases WHERE username = @username');
  const ins = d.prepare('INSERT INTO reviewed_cases (username, case_id) VALUES (@username, @case_id)');
  const tx = d.transaction(() => {
    del.run({ username });
    for (const case_id of caseIds) {
      ins.run({ username, case_id });
    }
  });
  tx();
}
```

Key design notes:
- All queries use named parameters (`@username`, `@case_id`) to prevent SQL injection
- `quality_flags` PK is (username, case_id, parameter) — a user can flag the same case on different parameters
- `saved_searches.filters` is stored as a JSON string (the CohortFilter object serialized)
- `setQualityFlags` / `setExcludedCases` / `setReviewedCases` use transactions: DELETE all then INSERT all (full-list replacement per assumption A3)
- `addSavedSearch` uses INSERT OR REPLACE for idempotency

**Part A: Create server/dataApi.ts**

Create an Express Router with 8 endpoints across 4 resources. All endpoints get the username from `req.auth!.preferred_username` (auth middleware already runs before /api/* routes).

```typescript
/**
 * Data API router: per-user CRUD for quality flags, saved searches,
 * excluded cases, and reviewed cases.
 *
 * All routes are protected by authMiddleware (global on /api/*).
 * Per D-06: data is per-user, scoped by req.auth.preferred_username.
 */

import { Router } from 'express';
import type { Request, Response } from 'express';
import {
  getQualityFlags, setQualityFlags,
  getSavedSearches, addSavedSearch, removeSavedSearch,
  getExcludedCases, setExcludedCases,
  getReviewedCases, setReviewedCases,
} from './dataDb.js';
import type { QualityFlagRow, SavedSearchRow } from './dataDb.js';

export const dataApiRouter = Router();

// ---------------------------------------------------------------------------
// Quality Flags (DATA-01)
// ---------------------------------------------------------------------------

dataApiRouter.get('/quality-flags', (req: Request, res: Response): void => {
  const username = req.auth!.preferred_username;
  const flags = getQualityFlags(username);
  // Map snake_case DB rows to camelCase for client
  const qualityFlags = flags.map((f) => ({
    caseId: f.case_id,
    parameter: f.parameter,
    errorType: f.error_type,
    flaggedAt: f.flagged_at,
    flaggedBy: f.flagged_by,
    status: f.status,
  }));
  res.json({ qualityFlags });
});

dataApiRouter.put('/quality-flags', (req: Request, res: Response): void => {
  const username = req.auth!.preferred_username;
  const { qualityFlags } = req.body as { qualityFlags?: unknown[] };

  if (!Array.isArray(qualityFlags)) {
    res.status(400).json({ error: 'qualityFlags array is required' });
    return;
  }

  // Map camelCase client data to snake_case DB rows
  const rows: QualityFlagRow[] = qualityFlags.map((f: any) => ({
    case_id: String(f.caseId ?? ''),
    parameter: String(f.parameter ?? ''),
    error_type: String(f.errorType ?? ''),
    flagged_at: String(f.flaggedAt ?? new Date().toISOString()),
    flagged_by: String(f.flaggedBy ?? username),
    status: String(f.status ?? 'open'),
  }));

  setQualityFlags(username, rows);

  // Return saved flags in camelCase
  const saved = getQualityFlags(username);
  res.json({
    qualityFlags: saved.map((f) => ({
      caseId: f.case_id,
      parameter: f.parameter,
      errorType: f.error_type,
      flaggedAt: f.flagged_at,
      flaggedBy: f.flagged_by,
      status: f.status,
    })),
  });
});

// ---------------------------------------------------------------------------
// Saved Searches (DATA-02)
// ---------------------------------------------------------------------------

dataApiRouter.get('/saved-searches', (req: Request, res: Response): void => {
  const username = req.auth!.preferred_username;
  const rows = getSavedSearches(username);
  const savedSearches = rows.map((r) => ({
    id: r.id,
    name: r.name,
    createdAt: r.created_at,
    filters: JSON.parse(r.filters),
  }));
  res.json({ savedSearches });
});

dataApiRouter.post('/saved-searches', (req: Request, res: Response): void => {
  const username = req.auth!.preferred_username;
  const { id, name, createdAt, filters } = req.body as Record<string, unknown>;

  if (typeof id !== 'string' || typeof name !== 'string') {
    res.status(400).json({ error: 'id and name are required' });
    return;
  }

  const row: SavedSearchRow = {
    id,
    name,
    created_at: typeof createdAt === 'string' ? createdAt : new Date().toISOString(),
    filters: typeof filters === 'object' ? JSON.stringify(filters) : '{}',
  };

  addSavedSearch(username, row);

  res.status(201).json({
    savedSearch: {
      id: row.id,
      name: row.name,
      createdAt: row.created_at,
      filters: JSON.parse(row.filters),
    },
  });
});

dataApiRouter.delete('/saved-searches/:id', (req: Request, res: Response): void => {
  const username = req.auth!.preferred_username;
  removeSavedSearch(username, req.params.id);
  res.json({ message: 'Saved search deleted' });
});

// ---------------------------------------------------------------------------
// Excluded Cases (DATA-03)
// ---------------------------------------------------------------------------

dataApiRouter.get('/excluded-cases', (req: Request, res: Response): void => {
  const username = req.auth!.preferred_username;
  res.json({ excludedCases: getExcludedCases(username) });
});

dataApiRouter.put('/excluded-cases', (req: Request, res: Response): void => {
  const username = req.auth!.preferred_username;
  const { excludedCases } = req.body as { excludedCases?: unknown[] };

  if (!Array.isArray(excludedCases)) {
    res.status(400).json({ error: 'excludedCases array is required' });
    return;
  }

  const caseIds = excludedCases.filter((c): c is string => typeof c === 'string');
  setExcludedCases(username, caseIds);
  res.json({ excludedCases: getExcludedCases(username) });
});

// ---------------------------------------------------------------------------
// Reviewed Cases (DATA-04)
// ---------------------------------------------------------------------------

dataApiRouter.get('/reviewed-cases', (req: Request, res: Response): void => {
  const username = req.auth!.preferred_username;
  res.json({ reviewedCases: getReviewedCases(username) });
});

dataApiRouter.put('/reviewed-cases', (req: Request, res: Response): void => {
  const username = req.auth!.preferred_username;
  const { reviewedCases } = req.body as { reviewedCases?: unknown[] };

  if (!Array.isArray(reviewedCases)) {
    res.status(400).json({ error: 'reviewedCases array is required' });
    return;
  }

  const caseIds = reviewedCases.filter((c): c is string => typeof c === 'string');
  setReviewedCases(username, caseIds);
  res.json({ reviewedCases: getReviewedCases(username) });
});
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client -> /api/data/* | All input arrays (qualityFlags, excludedCases, reviewedCases, savedSearches) come from untrusted client |
| user A -> user B data | Per-user isolation must prevent horizontal access — username comes from JWT, never from request body |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-08 | Information Disclosure | GET /api/data/* | mitigate | Username extracted from req.auth.preferred_username (JWT), never from query params; WHERE username = @username in all queries |
| T-04-09 | Tampering | PUT /api/data/* | mitigate | All SQL queries use named parameters (@username, @case_id) via better-sqlite3 prepared statements — no string concatenation |
| T-04-10 | Spoofing | /api/data/* auth bypass | mitigate | authMiddleware runs on all /api/* routes (mounted in index.ts before data routes); DATA-05 satisfied |
| T-04-11 | Denial of Service | PUT with large arrays | accept | express.json limit is 1mb; SQLite transactions handle bulk inserts efficiently; no pagination needed for v1 per-user data volumes |
| T-04-12 | Information Disclosure | saved_searches.filters JSON blob | accept | Filters contain search criteria (no PII); stored as opaque JSON; no server-side interpretation |
</threat_model>
