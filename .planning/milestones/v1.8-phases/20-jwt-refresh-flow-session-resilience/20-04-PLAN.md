    ```typescript
    /**
     * Phase 20 / D-02, D-09..D-12: client-side refresh state machine.
     * - Single-flight refreshPromise dedupes concurrent 401s within a tab.
     * - Retry-guard header X-EMD-Retry-After-Refresh prevents refresh loops.
     * - BroadcastChannel('emd-auth') coordinates across tabs.
     * - getCsrfFromCookie reads non-httpOnly emd-csrf cookie for double-submit.
     */

    const RETRY_GUARD_HEADER = 'X-EMD-Retry-After-Refresh';
    const REFRESH_URL = '/api/auth/refresh';

    // Module-level state
    let refreshPromise: Promise<string> | null = null;

    // BroadcastChannel — null in non-browser test environments lacking the global
    const bc: BroadcastChannel | null =
      typeof BroadcastChannel !== 'undefined' ? new BroadcastChannel('emd-auth') : null;

    bc?.addEventListener('message', (e) => {
      const msg = e.data as { type?: string; token?: string };
      if (msg?.type === 'refresh-success' && typeof msg.token === 'string') {
        sessionStorage.setItem('emd-token', msg.token);
      } else if (msg?.type === 'logout') {
        sessionStorage.removeItem('emd-token');
        if (!window.location.pathname.includes('/login')) window.location.href = '/login';
      }
    });

    function getCsrfFromCookie(): string {
      const match = /(?:^|;\s*)emd-csrf=([^;]+)/.exec(document.cookie);
      return match ? decodeURIComponent(match[1]) : '';
    }

    export function getAuthHeaders(): Record<string, string> {
      const token = sessionStorage.getItem('emd-token');
      return token ? { Authorization: `Bearer ${token}` } : {};
    }

    export function broadcastLogout(): void {
      bc?.postMessage({ type: 'logout' });
    }

    export async function refreshAccessToken(): Promise<string> {
      if (refreshPromise) return refreshPromise;
      refreshPromise = (async () => {
        try {
          const csrf = getCsrfFromCookie();
          const resp = await fetch(REFRESH_URL, {
            method: 'POST',
            credentials: 'include',
            headers: { 'X-CSRF-Token': csrf },
          });
          if (!resp.ok) throw new Error(`refresh_failed_${resp.status}`);
          const data = await resp.json() as { token: string; expiresAt: number };
          sessionStorage.setItem('emd-token', data.token);
          bc?.postMessage({ type: 'refresh-success', token: data.token, expiresAt: data.expiresAt });
          return data.token;
        } finally {
          refreshPromise = null;  // CRITICAL: reset on BOTH success and failure (Pitfall: "permanently blocks tab")
        }
      })();
      return refreshPromise;
    }

    function handleAuthFailure(): void {
      sessionStorage.removeItem('emd-token');
      bc?.postMessage({ type: 'logout' });
      if (!window.location.pathname.includes('/login')) window.location.href = '/login';
    }

    export async function authFetch(input: RequestInfo | URL, init?: RequestInit): Promise<Response> {
      const baseHeaders = (init?.headers ?? {}) as Record<string, string>;
      const headers = { ...getAuthHeaders(), ...baseHeaders };
      const resp = await fetch(input, { ...init, headers });

      if (resp.status !== 401) return resp;

      // Already a retry → do NOT refresh again (D-10 retry guard)
      if (baseHeaders[RETRY_GUARD_HEADER]) {
        handleAuthFailure();
        return resp;
      }

      // First 401 → attempt single-flight refresh
      try {
        const newToken = await refreshAccessToken();
        const retryHeaders = {
          ...baseHeaders,
          Authorization: `Bearer ${newToken}`,
          [RETRY_GUARD_HEADER]: '1',
        };
        return await fetch(input, { ...init, headers: retryHeaders });
      } catch {
        handleAuthFailure();
        return resp;
      }
    }
    ```

<threat_model>

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → /api/auth/refresh | Untrusted: cookie + CSRF cross here; client must include credentials:'include' AND X-CSRF-Token header |
| Browser sessionStorage → in-memory | XSS risk for access token; mitigated by sessionStorage scope (cleared on tab close) |
| BroadcastChannel('emd-auth') | Same-origin only by browser policy; no cross-origin attack surface |
| Cross-tab message → sessionStorage write | Trust: sibling tab's refresh-success blindly adopted; relies on same-origin BC contract |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-20-23 | DoS (refresh storm under load) | refreshAccessToken | mitigate | Module-level refreshPromise single-flight (D-11, Pattern 1, Pitfall 4); tested explicitly with 5 concurrent 401s asserting 1 refresh call |
| T-20-24 | Tampering (refresh loop after server token-version mismatch) | authFetch retry path | mitigate | X-EMD-Retry-After-Refresh header guard (D-10); retried request's 401 falls through to logout, no second refresh; tested |
| T-20-25 | Tampering (refreshPromise stays set after failure → tab permanently blocked) | refreshAccessToken finally block | mitigate | `refreshPromise = null` in `finally` block (RESEARCH anti-patterns explicit); tested via Test 9 (after failure, next 401 can refresh) |
| T-20-26 | Information Disclosure (refresh token readable from JS via document.cookie) | emd-refresh cookie | mitigate | httpOnly: true on emd-refresh (Plan 01); JS only reads emd-csrf (intentionally non-httpOnly for double-submit) |
| T-20-27 | Spoofing (forged BroadcastChannel message from XSS payload) | bc message handler | accept | Same-origin BC means an attacker would already have full origin access; refresh-success only updates sessionStorage (which the attacker also has access to). No additional attack surface |
| T-20-28 | DoS (logout request hangs → user can't sign out) | AuthContext.logout fetch | mitigate | Try/catch wraps server logout; local state cleared regardless; user UX never blocked by network |
| T-20-29 | Information Disclosure (CSRF token leaked to /api/auth/logout failure response body) | logout fetch | accept | Logout body is empty by design (Plan 01); no PII in CSRF header value (random hex) |
| T-20-30 | Repudiation (sibling tab uses adopted token after main tab forced logout) | BC message ordering | mitigate | logout broadcast precedes sessionStorage clear; sibling tabs receive {type:'logout'} which clears their own sessionStorage + redirects |

</threat_model>
