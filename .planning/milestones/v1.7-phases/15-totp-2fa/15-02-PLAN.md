Step D — Add POST /api/auth/totp/enroll handler. Append after the existing /change-password handler (around line 317). This route receives the enrollToken from /login in the body (NOT the Authorization header — it is in PUBLIC_PATHS). It re-signs a fresh enrollToken so the /confirm step uses a stable secret; actually the simpler pattern is to re-emit the SAME secret from the incoming enrollToken (the client already has it from /login). Implementation:

  authApiRouter.post('/totp/enroll', async (req: Request, res: Response): Promise<void> => {
    const { enrollToken } = req.body as { enrollToken?: string };
    if (typeof enrollToken !== 'string' || !enrollToken) {
      res.status(400).json({ error: 'enrollToken is required' });
      return;
    }
    let payload: { sub: string; purpose: string; totpSecret: string };
    try {
      payload = jwt.verify(enrollToken, getJwtSecret(), { algorithms: ['HS256'] }) as typeof payload;
    } catch {
      res.status(401).json({ error: 'Invalid or expired enrollToken' });
      return;
    }
    if (payload.purpose !== 'totp-enroll' || typeof payload.totpSecret !== 'string') {
      res.status(401).json({ error: 'Invalid enrollToken' });
      return;
    }
    const uri = authenticator.keyuri(payload.sub, 'EyeMatics', payload.totpSecret);
    const qrDataUrl = await QRCode.toDataURL(uri);
    // Re-issue a fresh token with same secret so clock resets to 3m from /enroll call
    const freshToken = signEnrollToken(payload.sub, payload.totpSecret);
    res.json({ qrDataUrl, manualKey: payload.totpSecret, enrollToken: freshToken });
  });

  authApiRouter.post('/totp/confirm', async (req: Request, res: Response): Promise<void> => {
    const { enrollToken, otp } = req.body as { enrollToken?: string; otp?: string };
    if (typeof enrollToken !== 'string' || typeof otp !== 'string' || !enrollToken || !otp) {
      res.status(400).json({ error: 'enrollToken and otp are required' });
      return;
    }
    let payload: { sub: string; purpose: string; totpSecret: string };
    try {
      payload = jwt.verify(enrollToken, getJwtSecret(), { algorithms: ['HS256'] }) as typeof payload;
    } catch {
      res.status(401).json({ error: 'Invalid or expired enrollToken' });
      return;
    }
    if (payload.purpose !== 'totp-enroll' || typeof payload.totpSecret !== 'string') {
      res.status(401).json({ error: 'Invalid enrollToken' });
      return;
    }
    if (!authenticator.check(otp, payload.totpSecret)) {
      res.status(401).json({ error: 'Invalid TOTP code' });
      return;
    }

    // Generate and hash 10 recovery codes concurrently (Pitfall 3)
    const rawCodes = Array.from({ length: 10 }, generateRecoveryCode);
    const hashedCodes = await Promise.all(rawCodes.map((c) => bcrypt.hash(c, 12)));

    await modifyUsers((users) =>
      users.map((u) =>
        u.username === payload.sub
          ? { ...u, totpSecret: payload.totpSecret, totpEnabled: true, totpRecoveryCodes: hashedCodes, lastLogin: new Date().toISOString() }
          : u,
      ),
    );

    const users = loadUsers();
    const user = users.find((u) => u.username === payload.sub);
    if (!user) {
      res.status(404).json({ error: 'User not found' });
      return;
    }

    // Mark audit action (follow existing res.locals.auditAction pattern from userCrud.ts)
    (res.locals as Record<string, string>).auditAction = 'totp-enrolled';

    const token = signSessionToken(user.username, user.role, user.centers);
    res.json({ token, recoveryCodes: rawCodes });
  });

| T-15-07 | Elevation of Privilege | POST /verify TOTP brute force | mitigate | Reuse existing rate limiter per username (shares counter with password attempts, already proven in T-02-06); 6-digit space × 3-period window = 3M combos, 5 attempts lockout limits exposure |
