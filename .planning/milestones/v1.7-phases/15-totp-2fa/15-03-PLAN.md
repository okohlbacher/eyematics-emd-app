    const startTotpEnroll = useCallback(async (): Promise<

      | { ok: false; error: string }
    > => {
      if (!pendingEnrollToken) return { ok: false, error: 'no_pending_token' };
      try {
        const resp = await fetch('/api/auth/totp/enroll', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ enrollToken: pendingEnrollToken }),
        });
        const data = await resp.json();
        if (!resp.ok) return { ok: false, error: data.error ?? 'enroll_failed' };
        setPendingEnrollToken(data.enrollToken); // refresh with fresh token
        return { ok: true, qrDataUrl: data.qrDataUrl, manualKey: data.manualKey, enrollToken: data.enrollToken };
      } catch {
        return { ok: false, error: 'network_error' };
      }
    }, [pendingEnrollToken]);

    /** Step 2 of TOTP enrollment: submit the 6-digit confirmation code. */
    const confirmTotpEnroll = useCallback(async (otp: string): Promise<
      { ok: true; recoveryCodes: string[]; token: string }
      | { ok: false; error: string }
    > => {
      if (!pendingEnrollToken) return { ok: false, error: 'no_pending_token' };
      try {
        const resp = await fetch('/api/auth/totp/confirm', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ enrollToken: pendingEnrollToken, otp }),
        });
        const data = await resp.json();
        if (!resp.ok) return { ok: false, error: data.error ?? 'confirm_failed' };
        // Do NOT clear requiresTotpEnrollment yet — wait for user to save recovery codes
        return { ok: true, recoveryCodes: data.recoveryCodes, token: data.token };
      } catch {
        return { ok: false, error: 'network_error' };
      }
    }, [pendingEnrollToken]);

    /** Called after user confirms they saved recovery codes — activates the session. */
    const completeTotpEnroll = useCallback((token: string) => {
      setRequiresTotpEnrollment(false);

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser→AuthContext state | All TOTP enrollment state lives in React memory; enrollToken is not written to localStorage |
| AuthContext→fetch to server | All requests go through same-origin fetch; cookies/auth headers per existing pattern |
| RecoveryCodesPanel→clipboard + Blob download | Plaintext recovery codes enter OS clipboard and user-downloaded file |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-15-17 | Information Disclosure | pendingEnrollToken written to localStorage | mitigate | Task 1: state kept in React useState only; NEVER persisted to localStorage/sessionStorage (enforce via code review — no `localStorage.setItem('enrollToken'` calls) |
| T-15-18 | Information Disclosure | Recovery codes leak via browser back button | accept | Once `completeTotpEnroll` fires the component unmounts; React state cleared; user cannot navigate back to a mounted RecoveryCodesPanel with codes |
| T-15-19 | Spoofing | TotpEnrollPage reachable without backend gate | mitigate | Task 3: App.tsx renders it ONLY when AuthContext.requiresTotpEnrollment is true; that flag is set only by /login response parsing (Task 1) |
| T-15-20 | Tampering | User bypasses checkbox gate in DevTools | accept | Continue button `disabled={!saved}` can be toggled client-side, but server has already activated enrollment in Plan 02 Task 1 and issued the session JWT; the checkbox is a UX safeguard, not a security control — the server-side recovery codes ARE active whether or not the user saved them |
| T-15-21 | Denial of Service | enrollToken expires during enrollment | mitigate | Task 1: startTotpEnroll refreshes the token from /enroll response (3-min fresh); Task 2 shows `totpEnrollErrorExpired` toast and instructs re-login |
| T-15-22 | Elevation of Privilege | Session token received before enrollment complete | mitigate | Task 2: RecoveryCodesPanel's Continue button is the single code path that calls completeTotpEnroll(token); no other caller can activate the session |

Residual risks:
- If the user closes the browser between /confirm success and Continue click, recovery codes are lost — per D-03 there is no regeneration path, so admin reset + re-enrollment is the recovery. Accepted per CONTEXT.md Deferred Ideas.
</threat_model>
